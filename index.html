class Ders:

def __init__(self, numara, ad):

self.numara = numara

self.ad = ad

self.ara_sinav_puani = 0

self.final_sinav_puani = 0

self.gecme_notu = 0

 

def notlari_gir(self):

self.ara_sinav_puani = float(input("Ara Sınav Puanı: "))

self.final_sinav_puani = float(input("Final Sınavı Puanı: "))

self.not_hesapla()

 

def not_hesapla(self):

self.gecme_notu = (self.ara_sinav_puani * 0.3) + (self.final_sinav_puani * 0.7)

self.durum = "Geçtiniz" if self.gecme_notu >= 50 else "Geçemediniz"

 

def ders_isimleri_listele(dersler):

print("{:<5} {:<40}".format("No", "Ders"))

for ders in dersler:

print("{:<5} {:<40}".format(ders.numara, ders.ad))

 

def notlari_goster(ders):

print("\nSeçilen Dersin Notları:")

print("{:<5} {:<40} {:<15} {:<15} {:<15} {:<15}".format("No", "Ders", "Ara Sınav", "Final Sınavı", "Geçme Notu", "Durum"))

print("{:<5} {:<40} {:<15} {:<15} {:<15} {:<15}".format(ders.numara, ders.ad, ders.ara_sinav_puani, ders.final_sinav_puani, ders.gecme_notu, ders.durum))

 

# Ders listesine her bir dersin numarası ve adı ekleniyor.

dersler = [

Ders(1, "TEMEL BİLGİ TEKNOLOJİLERİ I"),

Ders(2, "HALKLA İLİŞKİLER VE İLETİŞİM"),

Ders(3, "TEMEL SAĞLIK VE HASTALIK BİLGİSİ"),

Ders(4, "TIBBİ BELGELEME"),

Ders(5, "TIP TERİMLERİ")

]

 

while True:

ders_isimleri_listele(dersler)

secim = input("\nHangi dersin notlarını girmek istiyorsunuz? (Çıkış için 'q'): ")

 

if secim.lower() == 'q':

break

 

ders_bulundu = False

for ders in dersler:

if secim.isdigit() and int(secim) == ders.numara:

ders.notlari_gir()

notlari_goster(ders)

ders_bulundu = True

break

 

if not ders_bulundu:

print("Hatalı ders numarası, lütfen tekrar deneyin.")

 

# Sadece geçme notu açıklandıktan sonra mesajı göster

if ders_bulundu:

print("")

break
